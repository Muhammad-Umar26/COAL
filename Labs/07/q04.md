## Code:

```asm
include irvine32.inc

.data

.code
    main proc
        mov ecx, 5     ; using ecx as a variable to pass no. of columns
        call func

        exit
    main endp

    func proc
        mov ebx, ecx
        dec ebx

        mov edx, 1

        l1:
            push ecx

            cmp ebx, 0
            je skip

            mov ecx, ebx

            l2:
                mov eax, 32
                call writechar
            loop l2

            dec ebx
            
            skip:

            mov ecx, edx

            l3:
                mov eax, 42
                call writechar
            loop l3

            inc edx

            call crlf
            pop ecx
        loop l1

        ret
    func endp

end main
```
